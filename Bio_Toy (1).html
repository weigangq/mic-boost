def initialize(indiv_num,gene_num):
    population_pool=[]
    for i in range(indiv_num):
        individual=[]
        for j in range(gene_num):
            individual+=[np.random.randint(2)]
        population_pool+=[individual]
    return population_pool

def pop_mean_max_min(population):
    mysum=0
    mymax=0
    mymin=100000
    for individual in population:
        fitness=sum(individual)
        mysum+=fitness
        if fitness>mymax:
            mymax=fitness
        if fitness<mymin:
            mymin=fitness
    mymean=mysum/len(population)
    return mymean,mymax,mymin

def tournment(population):
    fitness_list=[sum(x) for x in population]
    bestone=fitness_list.index(max(fitness_list))
    return population[bestone]

def mutation(indiv,prob,num_gametes):
    next_pop=[]
    for i in range(num_gametes):
        next_indiv=[]
        for gene in indiv:
            if np.random.rand()<prob:
                next_indiv+=[1-gene]
            else:
                next_indiv+=[gene]
        next_pop+=[next_indiv]
    return next_pop
